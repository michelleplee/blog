        //For this project, we will make an LED that blinks while we hold the button down

// Pin 13 has an LED connected on most Arduino boards.
// give it a name:
int led = 13;

//define a value for the pushButton so that we can give it a new value later 
int switchState = 0;

// the setup routine runs once when you press reset:
void setup() {
  
  //let's have Serial begin so that we can talk to our computers
  Serial.begin(9600);
  
  // initialize the digital pin as an output.
  pinMode(led, OUTPUT);
  
  //initialize the pushbutton as an input
  pinMode(2, INPUT);

}

// the loop routine runs over and over again forever:
void loop() {
  
  //the value of the pushButton is now equal to the voltage that enters pin2
  switchState = digitalRead(2);
  
  
  if (switchState == HIGH) {
    //the button is pressed
    digitalWrite(led, HIGH);// turn the LED on (HIGH is the voltage level)               
    delay(1000);// wait for a second
    digitalWrite(led, LOW);
    delay(1000);// wait for a second
    Serial.println("working");
  } 
  
  else {
	//the button is not pressed
    digitalWrite(led, LOW);
  }
}

for(int pinNumber = 2; pinNumber<5; pinNumber++){
  pinMode(pinNumber, OUTPUT);
  digitalWrite(pinNumber, LOW);

  
}

}

void loop() {
  // put your main code here, to run repeatedly:
  int sensorVal = analogRead(sensorPin);
  Serial.print("Sensor Value: ");
  Serial.print(sensorVal);

  //convert the ADC reading to voltage
  float voltage = (sensorVal/1024.0) * 5.0;
  Serial.print(", Volts: ");
  Serial.print(voltage);

  Serial.print(", degrees C:");
  //convert the voltage to termperature in degrees
  float temperature = (voltage - .5) * 100;

  digitalWrite(2, LOW);

digitalWrite(2, HIGH);
